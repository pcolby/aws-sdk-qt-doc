98 /**
99  * @brief Create an AWS V4 Signature algorithm designation.
100  *
101  * This function returns an algorithm designation, as defined by Amazon, for use with
102  * V4 signatures.
103  *
104  * For example, if the algorith is `QCryptographicHash::Sha256`, this function will
105  * return `AWS4-HMAC-SHA256`.
106  *
107  * @param algorithm The hash algorithm to get the canonical designation for.
108  *
109  * @return An AWS V4 Signature algorithm designation.
110  *
111  * @see http://docs.aws.amazon.com/general/latest/gr/sigv4-create-string-to-sign.html
112  */
113 QByteArray AwsSignatureV4Private::algorithmDesignation(const QCryptographicHash::Algorithm algorithm) const
114 {
115  switch (algorithm) {
116  case QCryptographicHash::Md4: return "AWS4-HMAC-MD4";
117  case QCryptographicHash::Md5: return "AWS4-HMAC-MD5";
118  case QCryptographicHash::Sha1: return "AWS4-HMAC-SHA1";
119  case QCryptographicHash::Sha224: return "AWS4-HMAC-SHA224";
120  case QCryptographicHash::Sha256: return "AWS4-HMAC-SHA256";
121  case QCryptographicHash::Sha384: return "AWS4-HMAC-SHA384";
122  case QCryptographicHash::Sha512: return "AWS4-HMAC-SHA512";
123  default:
124  Q_ASSERT_X(false, Q_FUNC_INFO, "invalid algorithm");
125  return "invalid-algorithm";
126  }
127 }

129 /**
130  * @brief Create an AWS V4 Signature authorization header value.
131  *
132  * This function builds an V4 signature, and returns it to the caller. The returned
133  * header value is then suitable for adding as a `Authorization` header in the HTTP
134  * request, to be accepted by Amazon.
135  *
136  * @param credentials The AWS credentials to use to sign the request.
137  * @param operation The HTTP method being used for the request.
138  * @param request The network request to generate a signature for.
139  * @param payload Optional data being submitted in the request (eg for `PUT` and `POST` operations).
140  * @param timestamp The timestamp to use when signing the request.
141  *
142  * @return An AWS V4 Signature authorization header value.
143  *
144  * @see http://docs.aws.amazon.com/general/latest/gr/sigv4-signed-request-examples.html
145  * @see setAuthorizationHeader
146  */
147 QByteArray AwsSignatureV4Private::authorizationHeaderValue(const AwsAbstractCredentials &credentials,
148  const QNetworkAccessManager::Operation operation,
149  QNetworkRequest &request, const QByteArray &payload,
150  const QDateTime &timestamp) const
151 {
152  const QByteArray algorithmDesignation = this->algorithmDesignation(hashAlgorithm);
153  const AwsEndpoint endpoint(request.url().host());
154 
155  const QByteArray credentialScope = this->credentialScope(timestamp.date(), endpoint.regionName(), endpoint.serviceName());
156  QByteArray signedHeaders;
157  const QByteArray canonicalRequest = this->canonicalRequest(operation, request, payload, &signedHeaders);
158 
159  const QByteArray stringToSign = this->stringToSign(algorithmDesignation, timestamp, credentialScope, canonicalRequest);
160  const QByteArray signingKey = this->signingKey(credentials, timestamp.date(), endpoint.regionName(), endpoint.serviceName());
161  const QByteArray signature = QMessageAuthenticationCode::hash(stringToSign, signingKey, hashAlgorithm);
162 
163  return algorithmDesignation + " Credential=" + credentials.accessKeyId().toUtf8() + '/' + credentialScope +
164  ", SignedHeaders=" + signedHeaders + ", Signature=" + signature.toHex();
165 }

167 /**
168  * @brief Create an AWS V4 Signature canonical header string.
169  *
170  * In canonical form, header name and value are combined with a single semi-colon
171  * separator, with all whitespace removed from both, _except_ for whitespace within
172  * double-quotes.
173  *
174  * @param headerName Name of the HTTP header to convert to canonical form.
175  * @param headerValue Value of the HTTP header to convert to canonical form.
176  *
177  * @return An AWS V4 Signature canonical header string.
178  *
179  * @see http://docs.aws.amazon.com/general/latest/gr/sigv4-create-canonical-request.html
180  * @see canonicalHeaders
181  */
182 QByteArray AwsSignatureV4Private::canonicalHeader(const QByteArray &headerName, const QByteArray &headerValue) const
183 {
184  QByteArray header = headerName.toLower() + ':';
185  const QByteArray trimmedHeaderValue = headerValue.trimmed();
186  bool isInQuotes = false;
187  char previousChar = '\0';
188  for (int index = 0; index < trimmedHeaderValue.size(); ++index) {
189  char thisChar = trimmedHeaderValue.at(index);
190  header += thisChar;
191  if (isInQuotes) {
192  if ((thisChar == '"') && (previousChar != '\\'))
193  isInQuotes = false;
194  } else {
195  if ((thisChar == '"') && (previousChar != '\\')) {
196  isInQuotes = true;
197  } else if (isspace(thisChar)) {
198  while ((index < trimmedHeaderValue.size()-1) &&
199  (isspace(trimmedHeaderValue.at(index+1))))
200  ++index;
201  }
202  }
203  previousChar = thisChar;
204  }
205  return header;
206 }

208 /**
209  * @brief Create an AWS V4 Signature canonical headers string.
210  *
211  * This function constructs a canonical string containing all of the headers
212  * in the given request.
213  *
214  * @note \p request will typically not include a `Host` header at this stage,
215  * however Qt will add an appropriate `Host` header when the request is
216  * performed. So, if \p request does not include a `Host` header yet,
217  * this function will include a derived `Host` header in the canonical
218  * headers to allow for it.
219  *
220  * @param[in] request The network request to fetch the canonical headers from.
221  * @param[out] signedHeaders A semi-colon separated list of the names of all headers
222  * included in the result.
223  *
224  * @return An AWS V4 Signature canonical headers string.
225  *
226  * @see http://docs.aws.amazon.com/general/latest/gr/sigv4-create-canonical-request.html
227  * @see canonicalHeader
228  */
229 QByteArray AwsSignatureV4Private::canonicalHeaders(const QNetworkRequest &request, QByteArray * const signedHeaders) const
230 {
231  Q_CHECK_PTR(signedHeaders);
232  signedHeaders->clear();
233 
234  /* Note, Amazon says we should combine duplicate headers with comma separators...
235  * conveniently for us, QNetworkRequest requires that to have been done already.
236  * See note in QNetworkRequest::setRawHeader.
237  */
238 
239  // Convert the raw headers list to a map to sort on (lowercased) header names only.
240  QMap<QByteArray,QByteArray> headers;
241  foreach (const QByteArray &rawHeader, request.rawHeaderList()) {
242  headers.insert(rawHeader.toLower(), request.rawHeader(rawHeader));
243  }
244  // The "host" header is not included in QNetworkRequest::rawHeaderList, but will be sent by Qt.
245  headers.insert("host", request.url().host().toUtf8());
246 
247  // Convert the headers map to a canonical string, keeping track of which headers we've included too.
248  QByteArray canonicalHeaders;
249  for (QMap<QByteArray,QByteArray>::const_iterator iter = headers.constBegin(); iter != headers.constEnd(); ++iter) {
250  canonicalHeaders += canonicalHeader(iter.key(), iter.value()) + '\n';
251  if (!signedHeaders->isEmpty()) *signedHeaders += ';';
252  *signedHeaders += iter.key();
253  }
254  return canonicalHeaders;
255 }

257 /**
258  * @brief Create an AWS V4 Signature canonical request.
259  *
260  * @param[in] operation The HTTP method being used for the request.
261  * @param[in] request The network request to generate a canonical request for.
262  * @param[in] payload Optional data being submitted in the request (eg for `PUT` and `POST` operations).
263  * @param[out] signedHeaders A semi-colon separated list of the names of all headers
264  * included in the result.
265  *
266  * @return An AWS V4 Signature canonical request.
267  *
268  * @see http://docs.aws.amazon.com/general/latest/gr/sigv4-create-canonical-request.html
269  */
270 QByteArray AwsSignatureV4Private::canonicalRequest(const QNetworkAccessManager::Operation operation,
271  const QNetworkRequest &request, const QByteArray &payload,
272  QByteArray * const signedHeaders) const
273 {
274  return httpMethod(operation).toUtf8() + '\n' +
275  canonicalPath(request.url()).toUtf8() + '\n' +
276  canonicalQuery(QUrlQuery(request.url())) + '\n' +
277  canonicalHeaders(request, signedHeaders) + '\n' +
278  *signedHeaders + '\n' +
279  QCryptographicHash::hash(payload, hashAlgorithm).toHex();
280 }
