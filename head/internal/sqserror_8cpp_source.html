293 /**
294  * @brief Parse an SQS Error element.
295  *
296  * This function parses Error XML elements (ie from SQS ErrorResponse elements)
297  * as specified in http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
298  *
299  * For example:
300  *
301  * @code{xml}
302  * <Error>
303  * <Type>Sender</Type>
304  * <Code>AccessDenied</Code>
305  * <Message>Access to the resource http://sqs.us-east-1.amazonaws.com/ is denied.</Message>
306  * <Detail/>
307  * </Error>
308  * @endcode
309  *
310  * @param xml XML stream to parse.
311  *
312  * @see http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
313  */
314 void SqsErrorPrivate::parse(QXmlStreamReader &xml)
315 {
316  if ((xml.tokenType() == QXmlStreamReader::NoToken) && (!xml.readNextStartElement())) {
317  return;
318  }
319  Q_ASSERT(xml.name() == QLatin1String("Error"));
320  while (xml.readNextStartElement()) {
321  if (xml.name() == QLatin1String("Type")) {
322  type = typeFromString(rawType = xml.readElementText());
323  } else if (xml.name() == QLatin1String("Code")) {
324  code = codeFromString(rawCode = xml.readElementText());
325  } else if (xml.name() == QLatin1String("Message")) {
326  message = xml.readElementText();
327  } else if (xml.name() == QLatin1String("Detail")) {
328  detail = AwsAbstractResponse::toVariant(xml);
329  } else {
330  qWarning() << "ignoring" << xml.name();
331  xml.skipCurrentElement();
332  }
333  }
334 }

356  IfStringReturnErrorResponse(MissingAction);
357  IfStringReturnErrorResponse(MissingAuthenticationToken);
