53 void AwsSignatureV2::sign(const AwsAbstractCredentials &credentials, const QNetworkAccessManager::Operation operation,
54  QNetworkRequest &request, const QByteArray &data) const
55 {
56  Q_UNUSED(data) // Not included in V2 signatures.
57  Q_D(const AwsSignatureV2);
58 
59  // Set the AccessKeyId, SignatureMethod, SignatureVersion and Timestamp query items, if not already.
60  d->adornRequest(request, credentials);
61 
62  // Calculate the signature.
63  const QByteArray stringToSign = d->canonicalRequest(operation, request.url());
64  const QString signature = QString::fromUtf8(QUrl::toPercentEncoding(QString::fromUtf8(
65  QMessageAuthenticationCode::hash(stringToSign, credentials.secretKey().toUtf8(),
66  d->hashAlgorithm).toBase64())));
67 
68  // Append the signature to the request.
69  QUrl url = request.url();
70  url.setQuery(url.query() + QLatin1String("&Signature=") + signature);
71  request.setUrl(url);
72 }

107 /**
108  * @internal
109  *
110  * @brief Add AWS Signature Version 2 adornments to an AWS request.
111  *
112  * In addition to service-specific request parameters, Amazon requires that version
113  * 2 signatures contain a number of common query parameters. This functions adds
114  * those query parameters to \a request if they're not already present.
115  *
116  * The query parameters added by this function, as required by Amazon, are:
117  * * `AWSAccessKeyId` - set to \a credentials.accessKeyId().
118  * * `SignatureMethod` - set to `HMAC-SHA1` or `HMAC-SHA256`.
119  * * `SignatureVersion` - set to `2`.
120  * * `Timestamp` - set to a current UTC timestamp in an ISO 8601 format, like
121  * `2013-10-30T12:34:56Z`.
122  *
123  * @param request Request to adorn.
124  * @param credentials Credentials to use when adorning \a request.
125  *
126  * @see signatureMethod
127  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
128  */
129 void AwsSignatureV2Private::adornRequest(QNetworkRequest &request,
130  const AwsAbstractCredentials &credentials) const
131 {
132  // Set / add the necessary query items.
133  QUrl url = request.url();
134  QUrlQuery query(url);
135  setQueryItem(query, QLatin1String("AWSAccessKeyId"), credentials.accessKeyId());
136  setQueryItem(query, QLatin1String("SignatureVersion"), QLatin1String("2"));
137  setQueryItem(query, QLatin1String("SignatureMethod"), QString::fromUtf8(signatureMethod(hashAlgorithm)));
138  setQueryItem(query, QLatin1String("Timestamp"),
139  QString::fromUtf8(QUrl::toPercentEncoding(
140  QDateTime::currentDateTimeUtc().toString(QLatin1String("yyyy-MM-ddThh:mm:ssZ"))
141  )),
142  false); // Don't warn if its already set to something else.
143 
144  // If we've touched the query items (likely), then update the request.
145  if (query != QUrlQuery(url)) {
146  qDebug() << url;
147  url.setQuery(query);
148  qDebug() << url;
149  request.setUrl(url);
150  }
151 }

153 /**
154  * @internal
155  *
156  * @brief Create an AWS V2 Signature canonical request.
157  *
158  * This function creates a canonical representation of an AWS request as defined by
159  * Amazon's V2 signature specification.
160  *
161  * For example, for the following HTTP `GET` request:
162  *
163  * https://elasticmapreduce.amazonaws.com?Action=DescribeJobFlows&Version=2009-03-31&AWSAccessKeyId=AKIAIOSFODNN7EXAMPLE&SignatureVersion=2SignatureMethod=HmacSHA256Timestamp=2011-10-03T15%3A19%3A30
164  *
165  * this function will return the following canonical form:
166  *
167  * GET
168  * elasticmapreduce.amazonaws.com
169  * /
170  * AWSAccessKeyId=AKIAIOSFODNN7EXAMPLE&Action=DescribeJobFlows&SignatureMethod=HmacSHA256&SignatureVersion=2&Timestamp=2011-10-03T15%3A19%3A30&Version=2009-03-31
171  *
172  * @note All URL components are encoded to UTF-8, as required by Amazon.
173  *
174  * @param operation The HTTP method being requested.
175  * @param url The URL being request.
176  *
177  * @return An AWS V2 Signature canonical request.
178  *
179  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
180  */
181 QByteArray AwsSignatureV2Private::canonicalRequest(const QNetworkAccessManager::Operation operation,
182  const QUrl &url) const
183 {
184  return httpMethod(operation).toUtf8() + '\n' +
185  url.host().toUtf8() + '\n' +
186  canonicalPath(url).toUtf8() + '\n' +
187  canonicalQuery(QUrlQuery(url));
188 }

190 /**
191  * @brief Create an AWS V2 Signature method designation.
192  *
193  * This function returns a signature method designation, as defined by Amazon, for
194  * use with V2 signatures.
195  *
196  * For example, if the algorith is `QCryptographicHash::Sha256`, this function will
197  * return `HmacSHA256`.
198  *
199  * @note Amazon only supports two algorithms for V2 signatures - SHA1 and SHA256.
200  *
201  * @param algorithm The hash algorithm to get the canonical designation for.
202  *
203  * @return An AWS V2 Signature method designation.
204  *
205  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
206  */
207 QByteArray AwsSignatureV2Private::signatureMethod(const QCryptographicHash::Algorithm algorithm) const
208 {
209  switch (algorithm) {
210  case QCryptographicHash::Sha1: return "HmacSHA1";
211  case QCryptographicHash::Sha256: return "HmacSHA256";
212  default:
213  Q_ASSERT_X(false, Q_FUNC_INFO, "invalid algorithm");
214  return "invalid-algorithm";
215  }
