89 /**
90  * @internal
91  *
92  * @brief Create an AWS Signature version 1 canonical query.
93  *
94  * This function returns a string containing all non-empty query parameters in
95  * sorted order (case-insensitive), with no separators at all.
96  *
97  * For example, for the following SQS query string:
98  *
99  * ?Action=CreateQueue&QueueName=queue2&AWSAccessKeyId=AKIAIOSFODNN7EXAMPLE&SignatureVersion=1&Expires=2007-01-12T12:00:00Z&Version=2006-04-01
100  *
101  * this function will return the following canonical form:
102  *
103  * ActionCreateQueueAWSAccessKeyIdAKIAIOSFODNN7EXAMPLEExpires2007-01-12T12:00:00ZQueueNamequeue2SignatureVersion1Version2006-04-01
104  *
105  * This function is very similar to AwsAbstractSignature::canonicalQuery(), except
106  * that:
107  * 1. this function sorts case-insensitively, whereas AwsAbstractSignature::canonicalQuery()
108  * use a byte sort (ie is case sensitive); and
109  * 2. this function excludes parameters with empty values, where
110  * AwsAbstractSignature::canonicalQuery() includes all query parameters, regardless
111  * of content; and
112  * 3. this function does not use any separators in the generated string, whereas
113  * AwsAbstractSignature::canonicalQuery() uses `&` and `=` separators just as
114  * you would expect to see them in a typical query string; and
115  * 4. this function does not perform any URL encoding of the query parameters,
116  * whereas AwsAbstractSignature::canonicalQuery() URL encodes both parameter
117  * keys and values.
118  *
119  * The AwsAbstractSignature::canonicalQuery() function is used by the later signature
120  * algorithms, such as AwsSignatureV2 and AwsSignatureV4, as required by Amazon. Instead
121  * this function is specific to version 1 signatures.
122  *
123  * @param query Query to encode the HTTP query string from.
124  *
125  * @return An AWS Signature canonical query string.
126  *
127  * @see http://docs.aws.amazon.com/AmazonDevPay/latest/DevPayDeveloperGuide/LSAPI_Auth_REST.html#CalculatingHMACSignature
128  */
129 QByteArray AwsSignatureV1Private::canonicalQuery(const QUrlQuery &query) const
130 {
131  QList<QStringPair> list = query.queryItems(QUrl::FullyDecoded);
132  qSort(list.begin(), list.end(), AwsSignatureV1Private::caseInsensitiveLessThan);
133  QString result;
134  foreach (const QStringPair &pair, list) {
135  if (!pair.second.isEmpty()) {
136  result += pair.first + pair.second;
137  }
138  }
139  return result.toUtf8();
140 }

142 /**
143  * @internal
144  *
145  * @brief Is a key-value pair less than another key-value pair?
146  *
147  * This static function is used by the canonicalQuery function to sort query string
148  * parameters in case-insensitive order, via Qt's qSort function.
149  *
150  * @param pair1 The first key-value (query string parameter) pair.
151  * @param pair2 The second key-value (query string parameter) pair.
152  *
153  * @returns `true` if \a pair1 is less than \a pair2.
154  */
155 bool AwsSignatureV1Private::caseInsensitiveLessThan(const QStringPair &pair1, const QStringPair &pair2)
156 {
157  if (pair1.first.toLower() < pair2.first.toLower())
158  return true;
159  if (pair1.first.toLower() > pair2.first.toLower())
160  return false;
161  return (pair1.second.toLower() < pair2.second.toLower());
162 }
