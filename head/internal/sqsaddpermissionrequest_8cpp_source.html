265 QUrlQuery SqsAddPermissionRequestPrivate::urlQuery() const
266 {
267  QUrlQuery query = SqsRequestPrivate::urlQuery();
268  int index = 1; // SQS attributes are 1-indexed.
269  for (SqsAddPermissionRequest::PermissionsMap::const_iterator iter = permissions.cbegin();
270  iter != permissions.cend(); ++iter)
271  {
272  SqsAddPermissionRequest::PermissibleActions actions = iter.value();
273  if (!actions) continue;
274 
275  // Add the specified permissions.
276  #define IF_ACTION_SET_PARAMETER(action) \
277  if (actions.testFlag(SqsAddPermissionRequest::action##Action)) { \
278  while ((query.hasQueryItem(QString(AWS_ACCOUNT_ID_MEMBER_N).arg(index)))|| \
279  (query.hasQueryItem(QString( ACTION_NAME_MEMBER_N).arg(index)))) { \
280  ++index; \
281  } \
282  query.addQueryItem( \
283  QString(AWS_ACCOUNT_ID_MEMBER_N).arg(index), iter.key()); \
284  query.addQueryItem( \
285  QString(ACTION_NAME_MEMBER_N).arg(index), \
286  toString(static_cast<SqsRequest::Action>(SqsAddPermissionRequest::action##Action))); \
287  actions &= ~SqsAddPermissionRequest::action##Action; \
288  ++index; \
289  }
290  IF_ACTION_SET_PARAMETER(SendMessage)
291  IF_ACTION_SET_PARAMETER(ReceiveMessage)
292  IF_ACTION_SET_PARAMETER(DeleteMessage)
293  IF_ACTION_SET_PARAMETER(ChangeMessageVisibility)
294  IF_ACTION_SET_PARAMETER(GetQueueAttributes)
295  IF_ACTION_SET_PARAMETER(GetQueueUrl)
296  #undef IF_ACTION_SET_PARAMETER
297  Q_ASSERT(actions); // Sanity check.
298  }
299  return query;
300 }
