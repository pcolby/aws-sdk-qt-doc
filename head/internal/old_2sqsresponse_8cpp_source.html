65 /**
66  * @brief Get this response's error string.
67  *
68  * This override checks for SQS service errors, and if found, returns the first
69  * one. Otherwise, this override delegates to the base implementation (which
70  * checks for network and XML parse errors, for example).
71  *
72  * Also consider using the serviceErrors function to retrieve all of the servive
73  * errors (if any), not just the first one (though often SQS will only return
74  * one error per response).
75  *
76  * @return An error string, or a null QString if this response has no errors.
77  *
78  * @see AwsAbstractResponse::errorString
79  * @see hasError
80  * @see serviceErrors
81  */
82 QString SqsResponse::errorString() const
83 {
84  Q_D(const SqsResponse);
85  return (d->errors.empty()) ? AwsAbstractResponse::errorString() : d->errors.first().message();
86 }

122 /**
123  * @brief Parse a failure response.
124  *
125  * This implementation parses SQS ErrorResponse elements, as defined by
126  * http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
127  *
128  * For example:
129  * @code{xml}
130  * <ErrorResponse>
131  * <Error>
132  * <Type>Sender</Type>
133  * <Code>AccessDenied</Code>
134  * <Message>Access to the resource http://sqs.us-east-1.amazonaws.com/ is denied.</Message>
135  * <Detail/>
136  * </Error>
137  * <Error>
138  * ...
139  * </Error>
140  * <RequestId>9a285199-c8d6-47c2-bdb2-314cb47d599d</RequestId>
141  * </ErrorResponse>
142  * @endcode
143  *
144  * @param response Response to parse.
145  *
146  * @note Unlike most other SQS responses, the SQS ErrorResponse does not wrap
147  * the RequestId element in a ResponseMetadata element.
148  *
149  * @see http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
150  */
151 void SqsResponse::parseFailure(QIODevice &response)
152 {
153  Q_D(SqsResponse);
154  QXmlStreamReader xml(&response);
155  if (xml.readNextStartElement()) {
156  if (xml.name() == QLatin1String("ErrorResponse")) {
157  d->parseErrorResponse(xml);
158  } else {
159  qWarning() << "ignoring" << xml.name();
160  xml.skipCurrentElement();
161  }
162  }
163  setXmlError(xml);
164 }

189  *
190  * This implementation parses SQS ErrorResponse elements, as defined by
191  * http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
192  *
193  * For example:
194  * @code{xml}
195  * <ErrorResponse>
196  * <Error>

199  * <Message>Access to the resource http://sqs.us-east-1.amazonaws.com/ is denied.</Message>
200  * <Detail/>
201  * </Error>
202  * <Error>
203  * ...
204  * </Error>
205  * <RequestId>9a285199-c8d6-47c2-bdb2-314cb47d599d</RequestId>
206  * </ErrorResponse>
207  * @endcode
208  *
209  * @param xml XML stream to parse.
210  *
211  * @note Unlike most other SQS responses, the SQS ErrorResponse does not wrap
212  * the RequestId element in a ResponseMetadata element.
213  *
214  * @see http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
215  */
216 void SqsResponsePrivate::parseErrorResponse(QXmlStreamReader &xml)
217 {
218  Q_ASSERT(xml.name() == QLatin1String("ErrorResponse"));
219  while (xml.readNextStartElement()) {
220  if (xml.name() == QLatin1String("Error")) {
221  errors.append(SqsError(xml));
222  } else if (xml.name() == QLatin1String("RequestId")) {
223  requestId = xml.readElementText();
224  } else {
225  qWarning() << "ignoring" << xml.name();
226  xml.skipCurrentElement();
227  }
228  }
229 }
