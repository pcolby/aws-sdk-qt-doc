250 /**
251  * @brief Get an AwsRegion::Region enumeration value from a region name.
252  *
253  * @note The region name is evaluated in a case-insensitve manner.
254  *
255  * @param regionName AWS region name to get an enumeration value for.
256  *
257  * @return An AwsRegion::Region enumeration value corresponding to the specified
258  * AWS region name, or AwsRegion::InvalidRegion if the region name is not
259  * valid or not recognised.
260  */
261 AwsRegion::Region AwsRegion::fromName(const QString &regionName)
262 {
263  const QString lowerName(regionName.toLower());
264  if (lowerName == QLatin1String("ap-northeast-1")) return AP_Northeast_1;
265  if (lowerName == QLatin1String("ap-southeast-1")) return AP_Southeast_1;
266  if (lowerName == QLatin1String("ap-southeast-2")) return AP_Southeast_2;
267  if (lowerName == QLatin1String("eu-west-1")) return EU_West_1;
268  if (lowerName == QLatin1String("sa-east-1")) return SA_East_1;
269  if (lowerName == QLatin1String("us-east-1")) return US_East_1;
270  if (lowerName == QLatin1String("us-gov-west-1")) return US_Gov_West_1;
271  if (lowerName == QLatin1String("us-west-1")) return US_West_1;
272  if (lowerName == QLatin1String("us-west-2")) return US_West_2;
273  Q_ASSERT_X(false, Q_FUNC_INFO, qPrintable(QString::fromLatin1("invalid region name: %1").arg(regionName)));
274  return InvalidRegion;
275 }
