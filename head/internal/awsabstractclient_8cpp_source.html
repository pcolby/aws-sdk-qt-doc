162 /**
163  * @brief Send an AWS request.
164  *
165  * This base implementation simply uses the \a request's \c send method,
166  * supplying the necessary resources (ie the credentials, network access
167  * manager, and signature object) from this object as required.
168  *
169  * Derived classes would not normally need to override this method.
170  *
171  * @param request The AWS request to send.
172  *
173  * @return An object representing the AWS service's response.
174  */
175 AwsAbstractResponse * AwsAbstractClient::send(const AwsAbstractRequest &request)
176 {
177  Q_D(AwsAbstractClient);
178  Q_ASSERT(d->credentials);
179  Q_ASSERT(d->networkAccessManager);
180  Q_ASSERT(d->signature);
181  Q_ASSERT(request.isValid());
182 
183  if (!d->credentials) {
184  qWarning() << "credentials not set";
185  return NULL;
186  } else if (!d->networkAccessManager) {
187  qWarning() << "network access manager not set";
188  return NULL;
189  } else if (!d->signature) {
190  qWarning() << "signature not set";
191  return NULL;
192  } else if (!request.isValid()) {
193  qWarning() << serviceName() << "request not valid";
194  return NULL;
195  }
196 
197  return request.send(*d->networkAccessManager, endpoint(),
198  *d->signature, *d->credentials);
199 }

265 /**
266  * @brief Set the network access manager for this AWS service object.
267  *
268  * This object will not take ownership of the supplied \a manager pointer.
269  * It is the caller's responsibility to ensure that the pointer remains valid
270  * for the life of this object, or until setNetworkAccessManager is used to set
271  * a new pointer.
272  *
273  * @param manager Network access manager for this AWS service object to use.
274  */
275 void AwsAbstractClient::setNetworkAccessManager(QNetworkAccessManager * const manager)
276 {
277  Q_D(AwsAbstractClient);
278 
279  // If \a manager is already our network access manager, drop out early.
280  if (d->networkAccessManager == manager) {
281  return;
282  }
283 
284  // If the old network access manager was our parent, disown our parent.
285  if ((d->networkAccessManager != NULL) && (parent() == d->networkAccessManager)) {
286  setParent(NULL);
287  }
288 
289  // Set the network access manager.
290  d->networkAccessManager = manager;
291 }
