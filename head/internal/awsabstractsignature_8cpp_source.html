122 /**
123  * @brief Create an AWS Signature canonical path.
124  *
125  * This function simply returns the fully-URL-encoded path. However, if the path
126  * is empty, then a single '/' is returned, as is required by Amazon for both V2
127  * and V4 signatures (and presumably other versions too).
128  *
129  * @param url URL from which to extract the path.
130  *
131  * @return An AWS Signature canonical path.
132  *
133  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
134  * @see http://docs.aws.amazon.com/general/latest/gr/sigv4-create-canonical-request.html
135  */
136 QString AwsAbstractSignaturePrivate::canonicalPath(const QUrl &url) const
137 {
138  QString path = QDir::cleanPath(url.path(QUrl::FullyEncoded));
139  if (path.isEmpty()) {
140  return QString::fromLatin1("/");
141  }
142 
143  // If the path begins with "//", remove one of the redundant slashes.
144  // Note, this is only needed on Windows, because there QDir::speparator is
145  // '\', and internally QDir::cleanPath swaps all separators to '/', before
146  // calling qt_normalizePathSegments with allowUncPaths set to true, so that
147  // '//' is preserved to allow of Windows UNC paths beginning with '\\'.
148  // This should probably be reported as a bug in Qt::cleanPath("//...").
149 #ifdef Q_OS_WIN
150  if (path.startsWith(QLatin1String("//"))) {
151  path.remove(0, 1); // Remove the first of two forward slashes.
152  }
153 #endif
154 
155  // Restore the trailing '/' if QDir::cleanPath (rightly) removed one.
156  if ((url.path().endsWith(QLatin1Char('/'))) && (!path.endsWith(QLatin1Char('/')))) {
157  path += QLatin1Char('/');
158  }
159 
160  return path;
161 }

163 /**
164  * @brief Create an AWS Signature canonical query.
165  *
166  * This function returns an HTTP query string in Amazon's canonical form. That is,
167  * all query parameters are sorted by keys (**but not keys-then-values**), then
168  * joined with `&` separators, in `key=value` pairs with both keys and values being
169  * URL percent encoded.
170  *
171  * @note The canonical form produced by this function is used by Amazon's later
172  * signature formats (versions 2, 3 and 4), but not their earlier formats
173  * (versions 0 and 1).
174  *
175  * @param query Query to encode the HTTP query string from.
176  *
177  * @return An AWS Signature canonical query string.
178  *
179  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
180  * @see http://docs.aws.amazon.com/general/latest/gr/sigv4-create-canonical-request.html
181  */
182 QByteArray AwsAbstractSignaturePrivate::canonicalQuery(const QUrlQuery &query) const
183 {
184  typedef QPair<QString, QString> QStringPair;
185  QList<QStringPair> list = query.queryItems(QUrl::FullyDecoded);
186  qSort(list);
187  QString result;
188  foreach (const QStringPair &pair, list) {
189  if (!result.isEmpty()) result += QLatin1Char('&');
190  result += QString::fromUtf8(QUrl::toPercentEncoding(pair.first)) + QLatin1Char('=') +
191  QString::fromUtf8(QUrl::toPercentEncoding(pair.second));
192  }
193  return result.toUtf8();
194 }

196 /**
197  * @brief Create an AWS Signature request method string.
198  *
199  * This function simply converts QNetworkAccessManager operations (enum values)
200  * to strings appropriate to use in AWS signatures.
201  *
202  * @param operation The network operation to convert to string.
203  *
204  * @return A string representation of \p operation, or an empty string if the
205  * operation is not recognised or otherwise unsupported.
206  */
207 QString AwsAbstractSignaturePrivate::httpMethod(const QNetworkAccessManager::Operation operation) const
208 {
209  switch (operation) {
210  case QNetworkAccessManager::DeleteOperation: return QLatin1String("DELETE");
211  case QNetworkAccessManager::HeadOperation: return QLatin1String("HEAD");
212  case QNetworkAccessManager::GetOperation: return QLatin1String("GET");
213  case QNetworkAccessManager::PostOperation: return QLatin1String("POST");
214  case QNetworkAccessManager::PutOperation: return QLatin1String("PUT");
215  case QNetworkAccessManager::CustomOperation: // Fall through.
216  default:
217  // Catch this in debug mode for easier development / debugging.
218  Q_ASSERT_X(false, Q_FUNC_INFO, "invalid operation");
219  }
220  return QString(); // Operation was invalid / unsupported.
221 }

223 /**
224  * @brief Set a query item, checking for existing values first.
225  *
226  * This function is a light wrapper around QUrlQuery::addQueryItem() that first
227  * checks for existing values. Existing values will not be overwritten, instead
228  * if existing values are found, this function will simply check if the exsting
229  * value matches the desired \a value, and if not, will return `false` and
230  * optionally (according to \a warnOnNonIdenticalDuplicate) issue a qWarning().
231  *
232  * Typically, when setting something that must be a specific value, such as an
233  * access key ID, \a warnOnNonIdenticalDuplicate would be `true`. However, when
234  * setting query items as a fall-back default, such as a current timestamp,
235  * \a warnOnNonIdenticalDuplicate would typically be set to `false`.
236  *
237  * @param query URL query to add the query item to.
238  * @param key Query item key to add to \a query.
239  * @param value Query item value to add to \a query.
240  * @param warnOnNonIdenticalDuplicate If `true`, and an exisiting \a key
241  * value is found in \a query that has a value other than \a value,
242  * then a qWarning() is issued, otherwise the duplicate is silently
243  * ignored.
244  *
245  * @return `true` if the query item was set successfully or was already set to
246  * the requested value previously, `false` otherwise.
247  */
248 bool AwsAbstractSignaturePrivate::setQueryItem(QUrlQuery &query, const QString &key, const QString &value,
249  const bool warnOnNonIdenticalDuplicate) const
250 {
251  if (query.hasQueryItem(key)) {
252  const QString existingValue = query.queryItemValue(key, QUrl::FullyEncoded);
253  const bool existingQueryItemIsIdentical = (existingValue == value);
254  if ((warnOnNonIdenticalDuplicate) && (!existingQueryItemIsIdentical)) {
255  qWarning() << "AwsAbstractSignature::setQueryItem Not overwriting existing value for key"
256  << key << ':' << existingValue;
257  }
258  return existingQueryItemIsIdentical;
259  }
260  query.addQueryItem(key, value);
261  return true;
262 }
