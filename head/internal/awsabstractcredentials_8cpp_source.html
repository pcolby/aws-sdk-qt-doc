125 /**
126  * @brief Is this credentials object currently expired.
127  *
128  * In this base implementation, non-refreshable credentials (ie those for which
129  * isRefreshable() returns \c false) are assumed to never expire, whereas refreshable
130  * credentials are assume to be expired only if expiration() returns a valid QDateTime
131  * representing a time in the past (ie earlier than QDateTime::currentDateTimeUtc()).
132  *
133  * Derived classes may override this method to implement custom expiration logic.
134  *
135  * @returns \c true if this credentials object is expired, \c false otherwise.
136  *
137  * @see expiration()
138  * @see isRefreshable()
139  * @see refresh()
140  */
141 bool AwsAbstractCredentials::isExpired() const
142 {
143  if (isRefreshable()) {
144  const QDateTime expiration = this->expiration();
145  return ((expiration.isValid()) && (expiration < QDateTime::currentDateTimeUtc()));
146  }
147  return false; // Non-refreshable credentials do not expire, by default.
148 }
