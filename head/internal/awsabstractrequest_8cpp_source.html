32 /**
33  * @class AwsAbstractRequest
34  *
35  * @brief Interface class for providing AWS requests
36  *
37  * Typically, a user would pass a derived class (such as SqsCreateQueueRequest)
38  * to an AwsAbstractClient-derived class (such as SqsClient) to send a request
39  * to an AWS service.
40  *
41  * Note, by design, requests based on this class are endpoint-agnostic. That is,
42  * you may instantiate a request (such as an SqsCreateQueueRequest), and pass it
43  * to multiple clients (such as a US instance of SqsClient as well as an AU
44  * instance of SqsClient). Thus it is the caller (typically an
45  * AwsAbstractClient-derived class) that is responsible for for setting the
46  * region / endpoint information, allowing requests to be reused unmodified
47  * multiple times, and/or in multiple regions.
48  */

134 /**
135  * @brief Construct a QNetworkRequest for this AWS request.
136  *
137  * This base implementation used the protected unsignedRequest function to fetch
138  * the basic request to be sent, then uses \a signature and \a credentials to
139  * sign the request before returning to the caller.
140  *
141  * Derived classes may override this if their specific AWS service required more
142  * advanced logic (or disallow signing, for example).
143  *
144  * @param endpoint AWS endpoint the request is to be for.
145  * @param signature AWS signature class for signing the request.
146  * @param credentials AWS credentials to use for signing the request.
147  *
148  * @return A valid, signed (where appropriate) request that can be sent to via
149  * a network access manager to perform the configured request.
150  *
151  * @see unsignedRequest
152  */
153 QNetworkRequest AwsAbstractRequest::networkRequest(
154  const QUrl &endpoint,
155  const AwsAbstractSignature &signature,
156  const AwsAbstractCredentials &credentials) const
157 {
158  QNetworkRequest request(unsignedRequest(endpoint));
159  signature.sign(credentials, operation(), request, data());
160  return request;
161 }

179 /**
180  * @brief Send this request to a given Amazon endpoint, via the specified
181  * network access manager.
182  *
183  * @param manager Network access manager to send the request via.
184  * @param endpoint AWS service endpoint to send the request to.
185  * @param signature Signature implementation to use to sign the request.
186  * @param credentials AWS credentials to use to sign the request.
187  *
188  * @return AwsAbstractResponse-derived object representing the AWS service's response.
189  */
190 AwsAbstractResponse * AwsAbstractRequest::send(QNetworkAccessManager &manager,
191  const QUrl &endpoint,
192  const AwsAbstractSignature &signature,
193  const AwsAbstractCredentials &credentials) const
194 {
195  Q_ASSERT(isValid());
196  Q_D(const AwsAbstractRequest);
197  const QNetworkRequest request(networkRequest(endpoint, signature, credentials));
198  qDebug() << "Request URL:" << request.url().toString(QUrl::FullyEncoded);
199  if (request.hasRawHeader("Authorization")) {
200  qDebug() << "Authorization:" << request.rawHeader("Authorization");
201  }
202  switch (operation()) {
203  case QNetworkAccessManager::DeleteOperation:
204  return response(manager.deleteResource(request));
205  case QNetworkAccessManager::HeadOperation:
206  return response(manager.head(request));
207  case QNetworkAccessManager::GetOperation:
208  return response(manager.get(request));
209  case QNetworkAccessManager::PostOperation:
210  return response(d->post(manager, request));
211  case QNetworkAccessManager::PutOperation:
212  return response(d->put(manager, request));
213  case QNetworkAccessManager::CustomOperation: // Fall through.
214  default:
215  // Catch this in debug mode for easier development / debugging.
216  Q_ASSERT_X(false, Q_FUNC_INFO, "invalid operation");
217  }
218  return NULL;
219 }
