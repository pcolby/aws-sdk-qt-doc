85 /**
86  * @brief Get this response's error string.
87  *
88  * This base implementation returns either the internal network reply object's
89  * error string (if the reply object has an error), or the internal XML parse
90  * error (if one was set, eg via set setXmlError), otherwise a null QString.
91  *
92  * Derived classes may override this function to support additional error types.
93  * Typically such derived implementations would check this base implementation's
94  * result first, or fallback to this base implementation. For example:
95  *
96  * @code
97  * if (weHaveOurOwnCustomError()) {
98  * return customErrorString();
99  * } else {
100  * return AwsAbstractResponse::errorString();
101  * }
102  * @endcode
103  *
104  * @return An error string, or a null QString if this response has no errors.
105  *
106  * @see hasError
107  * @see networkError
108  * @see xmlParseError
109  */
110 QString AwsAbstractResponse::errorString() const
111 {
112  Q_D(const AwsAbstractResponse);
113  if (networkError() != QNetworkReply::NoError) {
114  return d->reply->errorString();
115  } else if (xmlParseError() != QXmlStreamReader::NoError) {
116  return xmlParseErrorString();
117  } else {
118  return QString();
119  }
120 }

123  * @brief Did any error occurr processing this reponse?
124  *
125  * Errors could be anything from network requests errors, to errors validating
126  * AWS response content.
127  *
128  * This base implementation supports network, and XML parsing errors. Derived
129  * classes may override this function to support additional error types.

145  return (((d->reply) && (d->reply->error() != QNetworkReply::NoError)) ||
146  (d->xmlError != QXmlStreamReader::NoError));

151  *
152  * This base implementation simply checks for the absense of network and XML

225 /**
226  * @brief Convert an XML stream to a hierarchical QVariantMap.
227  *
228  * This function is used internally to embed opaque XML structures, such as the
229  * SQS service's ErrorResponse::Error::Detail, which the SQS schema defines as
230  * an arbitrary complex type.
231  *
232  * @note This static function exists within the AwsAbstractResponse for
233  * historic reasons. It should probably be moved our of this class, and
234  * into a more generic utility space at some point.
235  *
236  * @param xml XML stream to convert.
237  * @param prefix Prefix to apply to special (non-element) child entry names.
238  * @param maxDepth Maximum depth to traverse before truncating the tree.
239  *
240  * @return A QVariantMap representing the \a xml XML fragment.
241  *
242  * @todo Move this toVariant function to somewhere more generic.
243  */
244 QVariantMap AwsAbstractResponse::toVariant(
245  QXmlStreamReader &xml, const QString &prefix, const int maxDepth)
246 {
247  if (maxDepth < 0) {
248  qWarning() << QObject::tr("max depth exceeded");
249  return QVariantMap();
250  }
251 
252  if (xml.hasError()) {
253  qWarning() << xml.errorString();
254  return QVariantMap();
255  }
256 
257  if (xml.tokenType() == QXmlStreamReader::NoToken)
258  xml.readNext();
259 
260  if ((xml.tokenType() != QXmlStreamReader::StartDocument) &&
261  (xml.tokenType() != QXmlStreamReader::StartElement)) {
262  qWarning() << QObject::tr("unexpected XML tokenType %1 (%2)")
263  .arg(xml.tokenString()).arg(xml.tokenType());
264  return QVariantMap();
265  }
266 
267  QVariantMap map;
268  if (xml.tokenType() == QXmlStreamReader::StartDocument) {
269  map.insert(prefix + QLatin1String("DocumentEncoding"), xml.documentEncoding().toString());
270  map.insert(prefix + QLatin1String("DocumentVersion"), xml.documentVersion().toString());
271  map.insert(prefix + QLatin1String("StandaloneDocument"), xml.isStandaloneDocument());
272  } else {
273  if (!xml.namespaceUri().isEmpty())
274  map.insert(prefix + QLatin1String("NamespaceUri"), xml.namespaceUri().toString());
275  foreach (const QXmlStreamAttribute &attribute, xml.attributes()) {
276  QVariantMap attributeMap;
277  attributeMap.insert(QLatin1String("Value"), attribute.value().toString());
278  if (!attribute.namespaceUri().isEmpty())
279  attributeMap.insert(QLatin1String("NamespaceUri"), attribute.namespaceUri().toString());
280  if (!attribute.prefix().isEmpty())
281  attributeMap.insert(QLatin1String("Prefix"), attribute.prefix().toString());
282  attributeMap.insert(QLatin1String("QualifiedName"), attribute.qualifiedName().toString());
283  map.insertMulti(prefix + attribute.name().toString(), attributeMap);
284  }
285  }
286 
287  for (xml.readNext(); (!xml.atEnd()) && (xml.tokenType() != QXmlStreamReader::EndElement)
288  && (xml.tokenType() != QXmlStreamReader::EndDocument); xml.readNext()) {
289  switch (xml.tokenType()) {
290  case QXmlStreamReader::Characters:
291  case QXmlStreamReader::Comment:
292  case QXmlStreamReader::DTD:
293  case QXmlStreamReader::EntityReference:
294  map.insertMulti(prefix + xml.tokenString(), xml.text().toString());
295  break;
296  case QXmlStreamReader::ProcessingInstruction:
297  map.insertMulti(prefix + xml.processingInstructionTarget().toString(),
298  xml.processingInstructionData().toString());
299  break;
300  case QXmlStreamReader::StartElement: {
301  const QString elementName = xml.name().toString();
302  map.insertMulti(elementName, toVariant(xml, prefix, maxDepth-1));
303  break;
304  }
305  default:
306  qWarning() << QObject::tr("unexpected XML tokenType %1 (%2)")
307  .arg(xml.tokenString()).arg(xml.tokenType());
308  }
309  }
310  return map;
311 }

313 /**
314  * @brief Does a network reply indicate a successful response?
315  *
316  * This base implementation simply checks that \a reply has no errors, and that
317  * it represents an HTTP response with a 2xx HTTP status code.
318  *
319  * Derived classes may override this function to perform other checks, and/or to
320  * support non-HTTP responses.
321  *
322  * This function is used by the base implementation of the parse function, to
323  * delegate responsibility to either the parseSuccess or parseFailure functions.
324  *
325  * @param reply Network reply to check for success.
326  *
327  * @return \c true if \a reply indicates a successful response, \c false otherwise.
328  *
329  * @see parse
330  */
331 bool AwsAbstractResponse::isSuccess(QNetworkReply * const reply) const
332 {
333  return ((reply->error() == QNetworkReply::NoError) &&
334  ((reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt() / 100) == 2));
335 }

389 /**
390  * @brief Parse an AWS response from a network reply.
391  *
392  * This base implementation reports networks, if any occurred, otherwise uses
393  * the result of the virtual isSuccess function to delegate parsing to either
394  * the pure virtual parseSuccess or pure virtual parseFailure methods.
395  *
396  * Typically derived classes will only implement parseSuccess and parseFailure,
397  * without needing to override this function.
398  *
399  * @param reply Network reply from which to parse the AWS response.
400  *
401  * @see isSuccess
402  * @see parseSuccess
403  * @see parseFailure
404  */
405 void AwsAbstractResponse::parse(QNetworkReply * const reply)
406 {
407  if (reply->error() != QNetworkReply::NoError) {
408  qWarning() << reply->errorString();
409  } else if (isSuccess(reply)) {
410  parseSuccess(*reply);
411  } else {
412  parseFailure(*reply);
413  }
414  emit finished();
415 }
