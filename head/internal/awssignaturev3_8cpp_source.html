127 /**
128  * @brief Create an AWS V3 Signature authorization header value.
129  *
130  * This function builds a V3 signature, and returns it to the caller. The returned
131  * header value is then suitable for adding as an `Authorization` header in the HTTP
132  * request, to be accepted by Amazon.
133  *
134  * @param credentials The AWS credentials to use to sign the request.
135  * @param operation The HTTP method being used for the request.
136  * @param request The network request to generate a signature for.
137  * @param payload Optional data being submitted in the request (eg for `PUT` and `POST` operations).
138  *
139  * @return An AWS V3 Signature authorization header value.
140  *
141  * @see http://docs.aws.amazon.com/Route53/latest/DeveloperGuide/RESTAuthentication.html
142  * @see setAuthorizationHeader
143  */
144 QByteArray AwsSignatureV3Private::authorizationHeaderValue(const AwsAbstractCredentials &credentials,
145  const QNetworkAccessManager::Operation operation,
146  QNetworkRequest &request, const QByteArray &payload) const
147 {
148  // Calculate the signature.
149  QByteArray signedHeaders;
150  QByteArray stringToSign = canonicalRequest(operation, request, payload, &signedHeaders);
151  if (!isHttps(request)) {
152  stringToSign = QCryptographicHash::hash(stringToSign, hashAlgorithm);
153  }
154  const QByteArray signature = QMessageAuthenticationCode::hash(
155  stringToSign, credentials.secretKey().toUtf8(), hashAlgorithm);
156 
157  // Build and return the authorization header value.
158  return
159  QByteArray((isHttps(request)) ? "AWS3-HTTPS " : "AWS3 ") +
160  "AWSAccessKeyId=" + credentials.accessKeyId().toUtf8() + ","
161  "Algorithm=" + algorithmDesignation(hashAlgorithm) + "," +
162  ((!isHttps(request)) ? "SignedHeaders=" + signedHeaders + ',' : "") +
163  "Signature=" + signature.toBase64();
164 }

166 /**
167  * @brief Create an AWS V3 Signature canonical header string.
168  *
169  * @note Amazon documentation does not specify how to handle whitespace within
170  * quotes for V3 signatures, so here we use the same approach as V3
171  * signatures. That is:
172  *
173  * In canonical form, header name and value are combined with a single semi-colon
174  * separator, with all whitespace removed from both, _except_ for whitespace within
175  * double-quotes.
176  *
177  * @note This function is only applicable to the `AWS3` format, not `AWS3-HTTPS`.
178  *
179  * @param headerName Name of the HTTP header to convert to canonical form.
180  * @param headerValue Value of the HTTP header to convert to canonical form.
181  *
182  * @return An AWS V3 Signature canonical header string.
183  *
184  * @see http://docs.aws.amazon.com/amazonswf/latest/developerguide/HMACAuth-swf.html
185  * @see http://docs.aws.amazon.com/general/latest/gr/sigV4-create-canonical-request.html
186  * @see canonicalHeaders
187  */
188 QByteArray AwsSignatureV3Private::canonicalHeader(const QByteArray &headerName, const QByteArray &headerValue) const
189 {
190  QByteArray header = headerName.toLower() + ':';
191  const QByteArray trimmedHeaderValue = headerValue.trimmed();
192  bool isInQuotes = false;
193  char previousChar = '\0';
194  for (int index = 0; index < trimmedHeaderValue.size(); ++index) {
195  char thisChar = trimmedHeaderValue.at(index);
196  header += thisChar;
197  if (isInQuotes) {
198  if ((thisChar == '"') && (previousChar != '\\'))
199  isInQuotes = false;
200  } else {
201  if ((thisChar == '"') && (previousChar != '\\')) {
202  isInQuotes = true;
203  } else if (isspace(thisChar)) {
204  while ((index < trimmedHeaderValue.size()-1) &&
205  (isspace(trimmedHeaderValue.at(index+1))))
206  ++index;
207  }
208  }
209  previousChar = thisChar;
210  }
211  return header;
212 }

214 /**
215  * @brief Create an AWS V3 Signature canonical headers string.
216  *
217  * This function constructs a canonical string containing all of the headers
218  * in the given request.
219  *
220  * @note \p request will typically not include a `Host` header at this stage,
221  * however Qt will add an appropriate `Host` header when the request is
222  * performed. So, if \p request does not include a `Host` header yet,
223  * this function will include a derived `Host` header in the canonical
224  * headers to allow for it.
225  *
226  * @note This function is only applicable to the `AWS3` format, not `AWS3-HTTPS`.
227  *
228  * @param[in] request The network request to fetch the canonical headers from.
229  * @param[out] signedHeaders A semi-colon separated list of the names of all headers
230  * included in the result.
231  *
232  * @return An AWS V3 Signature canonical headers string.
233  *
234  * @see http://docs.aws.amazon.com/general/latest/gr/sigV3-create-canonical-request.html
235  * @see canonicalHeader
236  */
237 QByteArray AwsSignatureV3Private::canonicalHeaders(const QNetworkRequest &request, QByteArray * const signedHeaders) const
238 {
239  Q_CHECK_PTR(signedHeaders);
240  signedHeaders->clear();
241 
242  /* Note, Amazon says we should combine duplicate headers with comma separators...
243  * conveniently for us, QNetworkRequest requires that to have been done already.
244  * See note in QNetworkRequest::setRawHeader.
245  */
246 
247  // Convert the raw headers list to a map to sort on (lowercased) header names only.
248  QMap<QByteArray,QByteArray> headers;
249  foreach (const QByteArray &rawHeader, request.rawHeaderList()) {
250  headers.insert(rawHeader.toLower(), request.rawHeader(rawHeader));
251  }
252  // The "host" header is not included in QNetworkRequest::rawHeaderList, but will be sent by Qt.
253  headers.insert("host", request.url().host().toUtf8());
254 
255  // Convert the headers map to a canonical string, keeping track of which headers we've included too.
256  QByteArray canonicalHeaders;
257  for (QMap<QByteArray,QByteArray>::const_iterator iter = headers.constBegin(); iter != headers.constEnd(); ++iter) {
258  // Only include "host" and "x-amz-*" headers. Note, Amazon documentation states that latter as
259  // "x-amz-" (ie with the trailing '-'), yet the official Amazon Java SDK tests for a "x-amz"
260  // prefix. Thus the Java SDK would include headers with keys like "x-amzfoo", but here we do not
261  // since that would disagree with the official documentation.
262  if ((iter.key() == "host") || (iter.key().startsWith("x-amz-"))) {
263  canonicalHeaders += canonicalHeader(iter.key(), iter.value()) + '\n';
264  if (!signedHeaders->isEmpty()) *signedHeaders += ';';
265  *signedHeaders += iter.key();
266  }
267  }
268  return canonicalHeaders;
269 }

271 /**
272  * @brief Create an AWS V3 Signature canonical request.
273  *
274  * Note, this function implments both `AWS3` and `AWS3-HTTPS` variants of the
275  * AWS Signature version 3 - which are quite different.
276  *
277  * @param[in] operation The HTTP method being used for the request.
278  * @param[in] request The network request to generate a canonical request for.
279  * @param[in] payload Optional data being submitted in the request (eg for `PUT` and `POST` operations).
280  * @param[out] signedHeaders A semi-colon separated list of the names of all headers
281  * included in the result.
282  *
283  * @return An AWS V3 Signature canonical request.
284  *
285  * @see http://docs.aws.amazon.com/amazonswf/latest/developerguide/HMACAuth-swf.html (AWS3)
286  * @see http://docs.aws.amazon.com/Route53/latest/DeveloperGuide/RESTAuthentication.html (AWS3-HTTPS)
287  */
288 QByteArray AwsSignatureV3Private::canonicalRequest(const QNetworkAccessManager::Operation operation,
289  const QNetworkRequest &request, const QByteArray &payload,
290  QByteArray * const signedHeaders) const
291 {
292  // AWS3-HTTPS
293  if (isHttps(request)) {
294  Q_ASSERT((request.hasRawHeader("x-amz-date")) || (request.hasRawHeader("Date")));
295  QByteArray canonicalRequest = request.rawHeader(request.hasRawHeader("x-amz-date") ? "x-amz-date" : "Date");
296  if (request.hasRawHeader("x-amz-nonce")) {
297  canonicalRequest += request.rawHeader("x-amz-nonce");
298  }
299  return canonicalRequest;
300  }
301 
302  // AWS3
303  return httpMethod(operation).toUtf8() + '\n' +
304  canonicalPath(request.url()).toUtf8() + '\n' +
305  canonicalQuery(QUrlQuery(request.url())) + '\n' +
306  canonicalHeaders(request, signedHeaders) + '\n' +
307  payload;
308 }
